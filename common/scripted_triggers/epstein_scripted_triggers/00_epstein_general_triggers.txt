custom_air_techs = {
	if = {
		limit = { has_dlc = "By Blood Alone" }
		has_game_rule = { rule = custom_air option = Custom }
	}
	else_if = {
		limit = { NOT = { has_dlc = "By Blood Alone" } }
		always = yes
	}
}
no_custom_air_techs = {
	if = {
		limit = { has_dlc = "By Blood Alone" }
		NOT = { has_game_rule = { rule = custom_air option = Custom } }
	}
	else_if = {
		limit = { NOT = { has_dlc = "By Blood Alone" } }
		always = no
	}
}
