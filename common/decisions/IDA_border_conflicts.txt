	IDA_border_conflict_montana = {
		state_target = yes
		target_trigger = {
		 	FROM = {
			any_neighbor_state = { is_owned_by = IDA }
			OR = {
				is_owned_by = MON
			}}
		}
		available = {
			has_completed_focus = IDA_bullock_raid
			has_war = no
			has_border_war = no
		}
		visible = {
			NOT = { 
				FROM = {
					is_owned_by = ROOT
				}
			}
		}
		cost = 75
		days_remove = 30
		days_re_enable = 365
		ai_will_do = {
			base = 25
		}
		complete_effect = {
			set_country_flag = planing_border_conflict
			hidden_effect = { MON = { country_event = { id = idaho.22 } } }
		}
		remove_effect = {
			FROM = { every_neighbor_state = { limit = {is_owned_by = IDA} ROOT = {
				start_border_war = {
					change_state_after_war =  no
    					attacker = {
        					state = PREV
        					num_provinces = 5
        					on_win = idaho.18
        					on_lose = idaho.20
       						on_cancel = idaho.20
    					}
				
    					defender = {
        					state = FROM
        					num_provinces = 5
        					on_win = idaho.21
        					on_lose = idaho.19
        					on_cancel = idaho.21
    					}
				}
			}}}
		}
	}
