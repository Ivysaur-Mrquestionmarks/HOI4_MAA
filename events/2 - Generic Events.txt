# ZZZ events #
country_event = {
	id = _zzz.1 #reunification ultiatum
	title = _zzz.1.t
	desc = _zzz.1.d
	#picture = 
	is_triggered_only = yes
	option = {
		name = _zzz.1.a #be a pussy and accept
		ai_chance = {
			factor = 1
			modifier = {
				factor = 2
				is_puppet_of = FROM
			}
		}
		FROM = {
			annex_country = {
    				target = ROOT
    				transfer_troops = yes
			}
		}
	}
	option = {
		name = _zzz.1.b #be a chad and denny
		ai_chance = {
			factor = 2
		}
		leave_faction = yes
		FROM = {
			set_autonomy = {
    				target = ROOT
    				autonomy_state = autonomy_free
			}
			create_wargoal = {
    				type = annex_everything
    				target = ROOT
			}
		}
	}
}

# Resource Events #
country_event = { #Import Rubber Event 1
	id = resources.1 
	title = resources.1.t
	desc = resources.1.d
	is_triggered_only = yes
	
	immediate = {
		if = {
			limit = {
				is_ai = yes
			}
			add_timed_idea = {
				idea = idea_ai_rubber_import_limit
				days = 180
			}
		}
	}
	
	option = { #Import 5 rubber
		name = resources.1.a 
		add_ideas = { idea_importing_rubber_5 }
		random_owned_controlled_state = {
			add_resource = {
				type = rubber
				amount = 5
			}
			set_state_flag = rubber_import_zone
		}
	}
	
	option = { #Import 10 rubber
		name = resources.1.b 
		add_ideas = { idea_importing_rubber_10 }
		random_owned_controlled_state = {
			add_resource = {
				type = rubber
				amount = 10
			}
			set_state_flag = rubber_import_zone
		}
	}
	
	option = { #Import 15 rubber
		name = resources.1.c
		add_ideas = { idea_importing_rubber_15 }
		random_owned_controlled_state = {
			add_resource = {
				type = rubber
				amount = 15
			}
			set_state_flag = rubber_import_zone
		}
	}
	
	option = { #Switch to more resource event
		name = resources.1.e
		ai_chance = {
			base = 1
			modifier = {
				factor = 0
				OR = {
					date < 2025.1.1 
      			}
			}
        }
		country_event = resources.2
	}
}
