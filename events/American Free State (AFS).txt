#End of The Civil War
country_event = {
	id = AFS.9
	title = AFS.9.t
	desc = AFS.9.d
	picture = event_
	is_triggered_only = no
	trigger = {
		OR = {
		    has_country_leader = {
                name = "Vermin Supreme"
                ruling_only = yes
            }
		    has_country_leader = {
                name = "Howie Hawkins"
                ruling_only = yes
            }
		    has_country_leader = {
                name = "James Mattis"
                ruling_only = yes
            }
	    }
		has_global_flag = ohio_civil_war_global
		has_war = no
	}
	mean_time_to_happen = { days = 0 }
	fire_only_once = yes
	immediate = {
    
	}
	option = {
		name = AFS.9.a
		ai_chance = {
			factor = 100
		}
		if = {
			limit = { has_country_flag = AFS_vermin_side }
			add_power_balance_value = {
				id = vermin_bop
				value = 0.15
				tooltip_side = vermin_powerful
			}
		}
		if = {
			limit = { has_country_flag = AFS_hawkins_side }
			add_power_balance_value = {
				id = hawkins_bop
				value = 0.15
				tooltip_side = hawkins_powerful
			}
		}
		if = {
			limit = { has_country_flag = AFS_mattis_side }
			add_power_balance_value = {
				id = mattis_bop
				value = 0.15
				tooltip_side = mattis_powerful
			}
		}
		if = {
			limit = {
		        has_country_leader = {
                    name = "Vermin Supreme"
                    ruling_only = yes
                }
		    }
			swap_ideas = {
				remove_idea = AFS_vermins_black_army_idea
				add_idea = AFS_vermins_black_army_weakened
			}
	    }
		if = {
			limit = {
		        has_country_leader = {
                    name = "Howie Hawkins"
                    ruling_only = yes
                }
		    }
			swap_ideas = {
				remove_idea = AFS_the_green_army
				add_idea = AFS_the_green_army_weakened
			}
	    }
		if = {
			limit = {
		        has_country_leader = {
                    name = "James Mattis"
                    ruling_only = yes
                }
		    }
			swap_ideas = {
				remove_idea = AFS_strength_and_might
				add_idea = AFS_strength_and_might_weakened
			}
	    }
		add_ideas = AFS_ruins_of_the_triumvirate
		add_ideas = AFS_a_scarred_population
		hidden_effect = {
			remove_ideas = AFS_peoples_support
			remove_ideas = AFS_peoples_support_2
			remove_ideas = AFS_vermin_sabotage
			remove_ideas = AFS_vermin_sabotage_2
			remove_ideas = AFS_peoples_support_3
			remove_ideas = AFS_legitimate_government
			remove_ideas = AFS_legitimate_government_2
			remove_ideas = AFS_the_unpopular_vermin
			remove_ideas = AFS_the_unpopular_vermin_2
			remove_ideas = AFS_the_unpopular_vermin_3
		    remove_global_flag = ohio_civil_war_global
		    set_country_flag = AFS_civil_war_over
		}
	}
}
