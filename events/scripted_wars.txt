add_namespace = reunification

#trump invades AFS
country_event = {
	id = reunification.1
	title = reunification.1.t
	desc = reunification.1.d
	#picture =
	fire_only_once = yes
	trigger = {
		tag = USA
		has_war = no
		date > 2023.6.6
		country_exists = AFS
		AFS = { has_war = no }
	}
	option = {
		name = reunification.1.a
		declare_war_on = {
		    target = AFS
		    type = annex_everything
		}
	}
}
